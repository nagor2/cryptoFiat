### Имплементация (смарт-контракт):

```solidity
// Пример vesting контракта для DFC

contract RLEVesting {
    IERC20 public rle;
    address public beneficiary;
    
    uint256 public cliff = 365 days;        // Timelock: 1 год
    uint256 public vestingDuration = 1460 days; // Vesting: 4 года
    uint256 public startTime;
    uint256 public totalAmount = 250_000 * 1e18; // 250k RLE
    uint256 public released;
    
    constructor(address _rle, address _beneficiary) {
        rle = IERC20(_rle);
        beneficiary = _beneficiary;
        startTime = block.timestamp;
    }
    
    function vestedAmount() public view returns (uint256) {
        if (block.timestamp < startTime + cliff) {
            // До cliff: 0 доступно
            return 0;
        }
        
        uint256 timeFromCliff = block.timestamp - (startTime + cliff);
        
        if (timeFromCliff >= vestingDuration) {
            // После vesting: всё доступно
            return totalAmount;
        }
        
        // Линейный vesting
        return (totalAmount * timeFromCliff) / vestingDuration;
    }
    
    function releasable() public view returns (uint256) {
        return vestedAmount() - released;
    }
    
    function release() external {
        require(msg.sender == beneficiary, "Not beneficiary");
        uint256 amount = releasable();
        require(amount > 0, "Nothing to release");
        
        released += amount;
        rle.transfer(beneficiary, amount);
    }
}
```

// contracts/LiquidityMining.sol
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.19;

import "@openzeppelin/contracts/token/ERC20/IERC20.sol";
import "@openzeppelin/contracts/security/ReentrancyGuard.sol";

contract LiquidityMining is ReentrancyGuard {
    IERC20 public stakingToken;  // LP токен (DFC/ETH от Uniswap)
    IERC20 public rewardsToken;  // RLE токен
    
    uint256 public rewardRate;   // Награда в секунду
    uint256 public lastUpdateTime;
    uint256 public rewardPerTokenStored;
    uint256 public periodFinish = 0;
    uint256 public rewardsDuration = 730 days; // 2 года
    
    mapping(address => uint256) public userRewardPerTokenPaid;
    mapping(address => uint256) public rewards;
    
    uint256 private _totalSupply;
    mapping(address => uint256) private _balances;
    
    // Events
    event Staked(address indexed user, uint256 amount);
    event Withdrawn(address indexed user, uint256 amount);
    event RewardPaid(address indexed user, uint256 reward);
    
    constructor(address _stakingToken, address _rewardsToken) {
        stakingToken = IERC20(_stakingToken);
        rewardsToken = IERC20(_rewardsToken);
    }
    
    function rewardPerToken() public view returns (uint256) {
        if (_totalSupply == 0) {
            return rewardPerTokenStored;
        }
        return
            rewardPerTokenStored +
            (((block.timestamp < periodFinish ? block.timestamp : periodFinish) 
              - lastUpdateTime) * rewardRate * 1e18) / _totalSupply;
    }
    
    function earned(address account) public view returns (uint256) {
        return
            ((_balances[account] * 
              (rewardPerToken() - userRewardPerTokenPaid[account])) / 1e18) +
            rewards[account];
    }
    
    // Stake LP tokens
    function stake(uint256 amount) external nonReentrant updateReward(msg.sender) {
        require(amount > 0, "Cannot stake 0");
        _totalSupply += amount;
        _balances[msg.sender] += amount;
        stakingToken.transferFrom(msg.sender, address(this), amount);
        emit Staked(msg.sender, amount);
    }
    
    // Withdraw LP tokens
    function withdraw(uint256 amount) public nonReentrant updateReward(msg.sender) {
        require(amount > 0, "Cannot withdraw 0");
        _totalSupply -= amount;
        _balances[msg.sender] -= amount;
        stakingToken.transfer(msg.sender, amount);
        emit Withdrawn(msg.sender, amount);
    }
    
    // Claim rewards
    function getReward() public nonReentrant updateReward(msg.sender) {
        uint256 reward = rewards[msg.sender];
        if (reward > 0) {
            rewards[msg.sender] = 0;
            rewardsToken.transfer(msg.sender, reward);
            emit RewardPaid(msg.sender, reward);
        }
    }
    
    // Exit: withdraw and claim
    function exit() external {
        withdraw(_balances[msg.sender]);
        getReward();
    }
    
    modifier updateReward(address account) {
        rewardPerTokenStored = rewardPerToken();
        lastUpdateTime = block.timestamp < periodFinish ? block.timestamp : periodFinish;
        if (account != address(0)) {
            rewards[account] = earned(account);
            userRewardPerTokenPaid[account] = rewardPerTokenStored;
        }
        _;
    }
}

// contracts/ProtocolRewardsClaim.sol
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.19;

import "@openzeppelin/contracts/token/ERC20/IERC20.sol";
import "@openzeppelin/contracts/utils/cryptography/MerkleProof.sol";

contract ProtocolRewardsClaim {
    IERC20 public rle;
    bytes32 public merkleRoot;
    address public owner;
    
    mapping(address => uint256) public claimed;
    
    event RewardsClaimed(address indexed user, uint256 amount);
    event MerkleRootUpdated(bytes32 newRoot, uint256 timestamp);
    
    constructor(address _rle) {
        rle = IERC20(_rle);
        owner = msg.sender;
    }
    
    // Пользователи claim rewards с Merkle proof
    function claim(
        uint256 totalAmount,
        bytes32[] calldata merkleProof
    ) external {
        require(claimed[msg.sender] < totalAmount, "Already claimed");
        
        // Verify Merkle proof
        bytes32 leaf = keccak256(abi.encodePacked(msg.sender, totalAmount));
        require(
            MerkleProof.verify(merkleProof, merkleRoot, leaf), 
            "Invalid proof"
        );
        
        uint256 claimable = totalAmount - claimed[msg.sender];
        claimed[msg.sender] = totalAmount;
        
        rle.transfer(msg.sender, claimable);
        emit RewardsClaimed(msg.sender, claimable);
    }
    
    // Admin обновляет Merkle root (еженедельно/ежемесячно)
    function updateMerkleRoot(bytes32 newRoot) external {
        require(msg.sender == owner, "Not owner");
        merkleRoot = newRoot;
        emit MerkleRootUpdated(newRoot, block.timestamp);
    }
    
    // View: сколько можно claim
    function getClaimable(address user, uint256 totalAmount) 
        external 
        view 
        returns (uint256) 
    {
        if (claimed[user] >= totalAmount) return 0;
        return totalAmount - claimed[user];
    }
}

### Off-Chain расчёт наград (Node.js скрипт)

```javascript
// scripts/calculateRewards.js
const { ethers } = require("hardhat");
const { MerkleTree } = require("merkletreejs");
const keccak256 = require("keccak256");

// Правила наград
const REWARDS = {
    CDP_OPEN: ethers.parseEther("10"),      // 10 RLE
    DEPOSIT_OPEN: ethers.parseEther("5"),   // 5 RLE
    AUCTION_BID: ethers.parseEther("2"),    // 2 RLE
};

async function calculateRewards() {
    const cdp = await ethers.getContractAt("CDP", CDP_ADDRESS);
    const deposit = await ethers.getContractAt("Deposit", DEPOSIT_ADDRESS);
    const auction = await ethers.getContractAt("Auction", AUCTION_ADDRESS);
    
    // 1. Читаем события за период
    const startBlock = LAST_UPDATE_BLOCK;
    const endBlock = await ethers.provider.getBlockNumber();
    
    const cdpOpens = await cdp.queryFilter(
        "PositionOpened", 
        startBlock, 
        endBlock
    );
    const depositOpens = await deposit.queryFilter(
        "DepositOpened", 
        startBlock, 
        endBlock
    );
    const auctionBids = await auction.queryFilter(
        "newBid", 
        startBlock, 
        endBlock
    );
    
    // 2. Считаем награды для каждого юзера
    const rewards = {};
    
    cdpOpens.forEach(event => {
        const user = event.args.owner;
        rewards[user] = (rewards[user] || 0n) + REWARDS.CDP_OPEN;
    });
    
    depositOpens.forEach(event => {
        const user = event.args.owner; // assuming owner field
        rewards[user] = (rewards[user] || 0n) + REWARDS.DEPOSIT_OPEN;
    });
    
    auctionBids.forEach(event => {
        const user = event.args.owner;
        rewards[user] = (rewards[user] || 0n) + REWARDS.AUCTION_BID;
    });
    
    // 3. Генерируем Merkle tree
    const leaves = Object.entries(rewards).map(([address, amount]) =>
        keccak256(
            ethers.solidityPacked(
                ["address", "uint256"],
                [address, amount]
            )
        )
    );
    
    const tree = new MerkleTree(leaves, keccak256, { sort: true });
    const root = tree.getHexRoot();
    
    // 4. Сохраняем proofs для пользователей
    const proofs = {};
    Object.entries(rewards).forEach(([address, amount]) => {
        const leaf = keccak256(
            ethers.solidityPacked(
                ["address", "uint256"],
                [address, amount]
            )
        );
        proofs[address] = {
            amount: amount.toString(),
            proof: tree.getHexProof(leaf)
        };
    });
    
    // 5. Публикуем JSON
    const output = {
        merkleRoot: root,
        totalRewards: Object.values(rewards).reduce((a, b) => a + b, 0n).toString(),
        claims: proofs,
        lastBlock: endBlock,
        timestamp: Date.now()
    };
    
    // Save to public JSON
    fs.writeFileSync(
        "./public/rewards-latest.json",
        JSON.stringify(output, null, 2)
    );
    
    console.log(`✅ Calculated rewards for ${Object.keys(rewards).length} users`);
    console.log(`📊 Total rewards: ${ethers.formatEther(output.totalRewards)} RLE`);
    console.log(`🌳 Merkle root: ${root}`);
    
    // 6. Update on-chain
    const rewardsClaim = await ethers.getContractAt(
        "ProtocolRewardsClaim", 
        REWARDS_CONTRACT
    );
    await rewardsClaim.updateMerkleRoot(root);
    
    console.log(`✅ Merkle root updated on-chain!`);
}

// Run weekly via cron
calculateRewards();
```
